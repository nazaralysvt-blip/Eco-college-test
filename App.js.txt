const titre = document.createElement("h3");
titre.innerText = a.nom;

const info = document.createElement("p");
info.innerText = `Places restantes : ${a.places - a.inscrits.length}`;

const btn = document.createElement("button");
btn.innerText = "S'inscrire";

btn.onclick = () => {
  if (a.inscrits.length >= a.places) {
    alert("Complet !");
    return;
  }

  const prenom = prompt("Prénom :");
  const classe = prompt("Classe :");

  if (!prenom || !classe) return;

  a.inscrits.push({ prenom, classe });
  sauvegarder();
  afficherActions();
};

card.appendChild(titre);
card.appendChild(info);
card.appendChild(btn);

div.appendChild(card);
